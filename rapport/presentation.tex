\begin{frame}{Détection des groupes}
	Groupes $\equiv$ composantes connexes du graphe :
	$$N'(t)=\bigcup_i N'_i(t).$$

	Rappel : $N'_i(t) = \{j \in N_i, ||x_i - x_j|| < R \rho^t\}.$
\end{frame}
